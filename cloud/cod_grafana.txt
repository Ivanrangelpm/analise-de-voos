#!/bin/bash

# --- PASSO 1: Atualizar o sistema e instalar dependências ---
echo "=> Atualizando o sistema e instalando dependências..."
sudo apt-get update
sudo apt-get upgrade -y
sudo apt-get install -y apt-transport-https software-properties-common wget gpg

# --- PASSO 2: Adicionar a chave GPG e o repositório do Grafana (Método Moderno) ---
echo "=> Adicionando o repositório do Grafana..."
sudo mkdir -p /etc/apt/keyrings/
wget -q -O - https://packages.grafana.com/gpg.key | gpg --dearmor | sudo tee /etc/apt/keyrings/grafana.gpg > /dev/null
echo "deb [signed-by=/etc/apt/keyrings/grafana.gpg] https://packages.grafana.com/oss/deb stable main" | sudo tee /etc/apt/sources.list.d/grafana.list

# --- PASSO 3: Instalar e iniciar o Grafana ---
echo "=> Instalando e iniciando o Grafana..."
sudo apt-get update
sudo apt-get install grafana -y
sudo systemctl start grafana-server
sudo systemctl enable grafana-server

# --- PASSO 4: Verificar o status e finalizar ---
echo "=> Verificação final do serviço:"
sudo systemctl status grafana-server --no-pager
echo "--------------------------------------------------"
echo "✅ Grafana instalado com sucesso!"
echo "Acesse o IP público na porta 3000. Ex: http://SEU_IP_PUBLICO:3000"
echo "Usuário padrão: admin"
echo "Senha padrão: admin (você deverá trocá-la no primeiro login)"
